```go
func lengthOfLongestSubstring(s string) int {
    // 哈希map记录字符是否出现过
    m := map[byte]int{}
    n := len(s)

    rk, ans := -1, 0 // 右指针和无重复子串长度

    // 左指针从0开始遍历
    for i := 0; i < n; i++ {
        if i != 0 {
            // 左指针向右移动一格，删除左指针原来字符
            delete(m, s[i-1])
        }
        // 当右指针没扫完字符串或没遇到重复字符时
        for rk + 1 < n && m[s[rk+1]] == 0 {
            //标记当前字符已存在，右指针右移
            m[s[rk+1]]++
            rk++
        }

        // 更新最长子串长度
        ans = max(ans, rk - i + 1)
    }
    return ans
}

func max(x, y int) int {
    if x < y {
        return y
    }
    return x
}
```

### 146 LRU缓存机制

双链表结合map实现缓存，最新使用的节点置于头结点，则LRU的节点为尾节点

```go
// 双向链表存储键值对
type DLinkedNode struct {
    key, value int
    prev, next *DLinkedNode
}

// 缓存结构体
type LRUCache struct {
    size int
    capacity int
    cache map[int]*DLinkedNode // 值是一个包含键值对信息的链表节点
    head, tail *DLinkedNode // 哨兵头尾，不含键值对
}

// 初始化新节点
func initDLinkedNode(key, value int) *DLinkedNode {
    return &DLinkedNode{
        key: key
        value: value
    }
}

// 初始化缓存
func Constructor(capacity int) LRUCache {
    l := LRUCache{
        capacity: capacity,
        cache: map[int]*DLinkedNode{},
        head: initDLinkedNode(0, 0),
        tail: initDLinkedNode(0, 0),
    }

    l.head.next = l.tail
    l.tail.prev = l.head
    return l
}

// 节点加到链表头部
func (this *LRUCache) addToHead(node *DLinkedNode) {
    node.prev = this.head
    node.next = this.head.next
    this.head.next.prev = node
    this.head.next = node
}

// 摘除节点
func (this *LRUCache) removeNode(node *DLinkedNode) {
    node.prev.next = node.next
    node.next.prev = node.prev
}

// 先把节点从原位置删掉，再重新插到头部
func (this *LRUCache) moveTohead(node *DLinkedNode) {
    this.removeNode(node)
    this.addToHead(node)
}

// 删除尾节点(不删哨兵尾巴，而是删除最后一个键值对节点)
func (this *LRUCache) removeTail() *DLinkedNode {
    node := this.tail.prev
    this.removeNode(node)
    return node
}

// 查询
func (this *LRUCache) Get(key int) int {
    if _, ok := this.cache[key]; !ok {
        return -1
    }

    node := this.cache[key]
    this.moveTohead(node)
    return node.value
}

// 插入新节点
func (this *LRUCache) Put(key int, value int)  {
    if _,ok := this.cache[key]; !ok{ //不存在的节点则初始化加入
        node := initDLinkedNode(key,value)
        this.cache[key] = node
        this.addToHead(node)
        this.size++
        if this.size > this.capacity {
            removed := this.removeTail() // 在链表中摘除这个节点后再在cache中释放
            delete(this.cache, removed.key)
            this.size--
        }
    } else { //已存在的节点则更新值并置顶
        node := this.cache[key]
        node.value = value
        this.moveTohead(node)
    }
    
}


/**
 * Your LRUCache object will be instantiated and called as such:
 * obj := Constructor(capacity);
 * param_1 := obj.Get(key);
 * obj.Put(key,value);
 */
```

### 215 数组中第K大的元素

方法一：快排

```go
func findKthLargest(nums []int, k int) int {
    n := len(nums)
    return quickselect(nums, 0, n-1, n-k) //数组，左边界，右边界，第n-k即为第k大的数
}

func quickselect(nums []int, l, r, k int) int {
    //整个递归的逻辑是：不断缩小包含目标下标 k 的区间，直到区间缩小到只有一个元素（l == r），这个元素必然是 nums [k]。如果 l == r，说明 l 和 r 都等于 k
    if (l == r){
        return nums[k] //左右边界相等说明说明只剩一个元素，就是目标元素
    }

    partition := nums[l] //
    //先把左右指针放在两侧是避免partition是最大、最小值导致指针越界
    i := l - 1
    j := r + 1
    for (i < j) {
        // 左指针右移找大数；右指针左移找小数
        for i++; nums[i]<partition; i++{}
        for j--; nums[i]>partition; i--{}
        if(i < j){
            nums[i], nums[j] = nums[j], nums[i]
        }
    }

    // 分区完成后，j 是分区的分界点：[l,j] 都 <= 基准值，[j+1,r] 都 >= 基准值
    // 如果目标下标 k 小于等于 j，说明目标元素在左区间，递归处理左区间
    if(k <= j) {
        return quickselect(nums, l, j, k)
    }else{
        return quickselect(nums, j+1, r, k)
    }
```

```go
// findKthLargest 是主函数，通过大顶堆实现找数组中第 k 大的元素
func findKthLargest(nums []int, k int) int {
    // 初始化堆的大小为数组长度（初始时整个数组作为堆）
    heapSize := len(nums)
    // 第一步：将数组构建成一个大顶堆（堆顶是当前最大元素）
    buildMaxHeap(nums, heapSize)
    // 第二步：循环 k-1 次，每次把堆顶的最大元素交换到数组末尾，然后调整堆
    // 循环范围：i 从数组最后一位，到 len(nums)-k+1（共 k-1 次交换）
    for i := len(nums) - 1; i >= len(nums) - k + 1; i-- {
        // 交换堆顶元素（nums[0]，当前最大）和当前未排序部分的最后一位（nums[i]）
        // 相当于把最大元素"移除"堆（放到数组末尾）
        nums[0], nums[i] = nums[i], nums[0]
        // 堆大小减 1，因为末尾的元素已经是排序好的，不再参与堆调整
        heapSize--
        // 交换后堆顶元素被替换，需要重新调整堆，恢复大顶堆的性质
        maxHeapify(nums, 0, heapSize)
    }
    // 经过 k-1 次交换后，堆顶元素（nums[0]）就是第 k 大的元素
    return nums[0]
}

// buildMaxHeap 构建大顶堆：从最后一个非叶子节点开始，自下而上调整堆
// 参数说明：
// a: 待构建堆的数组
// heapSize: 当前堆的有效大小（数组前 heapSize 个元素是堆）
func buildMaxHeap(a []int, heapSize int) {
    // 最后一个非叶子节点的下标：heapSize/2 - 1
    // 从该节点向前遍历到根节点（下标 0），逐个调整堆
    for i := heapSize / 2 - 1; i >= 0; i-- {
        // 对下标 i 的节点进行堆调整，保证以 i 为根的子树是大顶堆
        maxHeapify(a, i, heapSize)
    }
}

// maxHeapify 核心堆调整函数：保证以 i 为根的子树满足大顶堆性质（父节点 >= 子节点）
// 参数说明：
// a: 堆数组
// i: 需要调整的节点下标（当前子树的根）
// heapSize: 当前堆的有效大小
func maxHeapify(a []int, i, heapSize int) {
    // 初始化三个变量：
    // l: i 的左子节点下标（i*2 + 1）
    // r: i 的右子节点下标（i*2 + 2）
    // largest: 记录 i、l、r 中值最大的节点下标，初始为 i
    l, r, largest := i*2+1, i*2+2, i
    // 1. 比较左子节点和当前根节点：如果左子节点更大，更新 largest 为左子节点下标
    if l < heapSize && a[l] > a[largest] {
        largest = l
    }
    // 2. 比较右子节点和当前 largest 节点：如果右子节点更大，更新 largest 为右子节点下标
    if r < heapSize && a[r] > a[largest] {
        largest = r
    }
    // 3. 如果 largest 不是 i（说明子节点中有更大的值）
    if largest != i {
        // 交换根节点 i 和 largest 节点的值，让更大的值成为父节点
        a[i], a[largest] = a[largest], a[i]
        // 交换后，largest 位置的子树可能不再满足堆性质，递归调整该子树
        maxHeapify(a, largest, heapSize)
    }
}
```

### 25 K个一组翻转链表

1. 核心思路：分批次处理链表，每 k 个节点为一组，先找组的尾节点，翻转该区间后重新衔接前后链表；

2. 关键指针：

   - `hair`：虚拟头节点，统一处理头节点翻转；
   - `pre`：每组的前驱节点，负责衔接翻转后的组头；
   - `tail`：每组的尾节点，用于判断是否剩余足够节点，也是翻转的终止边界；

   

3. 翻转逻辑：`myReverse` 函数通过「前驱指针 prev + 遍历指针 p」实现区间内节点的反向指向，最终返回翻转后的头尾节点。

简单记：「找区间→翻转区间→衔接区间」三步循环，虚拟头节点解决头节点处理难题。

```go
// ListNode 是链表节点的定义（题目隐含的基础结构，补充出来方便理解）
// type ListNode struct {
//     Val int
//     Next *ListNode
// }

// reverseKGroup 是主函数，实现「K 个一组翻转链表」的核心逻辑
// 参数：
//   head: 原始链表的头节点
//   k: 每组需要翻转的节点数
// 返回值：翻转后的链表头节点
func reverseKGroup(head *ListNode, k int) *ListNode {
    // 1. 创建「虚拟头节点 hair」（哨兵节点），Next 指向原始头节点 head
    // 作用：统一处理链表头部的翻转（避免单独处理头节点的边界问题）
    hair := &ListNode{Next: head}
    // 2. pre 指针：记录每一组待翻转区间的「前驱节点」，初始指向虚拟头节点
    pre := hair

    // 3. 主循环：只要当前组的头节点 head 不为空，就继续处理
    for head != nil {
        // 4. tail 指针：找到当前组的「尾节点」，初始指向 pre（前驱节点）
        tail := pre
        // 5. 循环 k 次，让 tail 移动到当前组的最后一个节点
        for i := 0; i < k; i++ {
            tail = tail.Next
            // 6. 如果中途 tail 为空，说明剩余节点不足 k 个，直接返回结果（不翻转）
            if tail == nil {
                return hair.Next
            }
        }
        // 7. 记录当前组尾节点的下一个节点（后续需要衔接翻转后的链表）
        nex := tail.Next
        // 8. 调用 myReverse 翻转 [head, tail] 区间内的节点，返回翻转后的新头、新尾
        head, tail = myReverse(head, tail)
        // 9. 衔接前驱节点：pre 的 Next 指向翻转后的新头节点
        pre.Next = head
        // 10. 衔接后继节点：翻转后的新尾节点的 Next 指向之前记录的 nex
        tail.Next = nex
        // 11. 更新 pre：作为下一组的前驱节点（当前组的尾节点）
        pre = tail
        // 12. 更新 head：下一组的头节点（当前组尾节点的下一个节点）
        head = tail.Next
    }
    // 13. 所有组处理完成，返回虚拟头节点的 Next（翻转后的链表真正头节点）
    return hair.Next
}

// myReverse 翻转链表中 [head, tail] 区间内的所有节点
// 参数：
//   head: 待翻转区间的头节点
//   tail: 待翻转区间的尾节点
// 返回值：翻转后的区间头节点（原 tail）、区间尾节点（原 head）
func myReverse(head, tail *ListNode) (*ListNode, *ListNode) {
    // 1. prev 指针：初始指向 tail.Next（区间外的第一个节点），作为翻转的「前驱节点」
    // 作用：让翻转后的节点 Next 指向这个节点，避免链表断裂
    prev := tail.Next
    // 2. p 指针：遍历待翻转区间的节点，初始指向 head
    p := head
    // 3. 循环：直到 prev 移动到 tail 位置（所有节点翻转完成）
    for prev != tail {
        // 4. 记录 p 的下一个节点（避免翻转时丢失后续节点）
        nex := p.Next
        // 5. 翻转 p 的 Next 指针：指向 prev（实现节点反向）
        p.Next = prev
        // 6. prev 向前移动：指向当前 p 节点（作为下一个节点的前驱）
        prev = p
        // 7. p 向前移动：指向之前记录的 nex 节点
        p = nex
    }
    // 8. 翻转完成后，原 tail 变成区间头，原 head 变成区间尾，返回这两个节点
    return tail, head
}
```

### 15 三数之和

```go
import "sort"

// threeSum 找出数组中所有不重复的三元组 [a, b, c]，满足 a + b + c = 0
func threeSum(nums []int) [][]int {
    // 获取数组长度
    n := len(nums)
    // 核心步骤：排序！排序后才能用双指针，并且方便去重
    sort.Ints(nums)
    // 初始化结果切片，用于存储所有符合条件的三元组
    ans := make([][]int, 0)

    // 最外层循环：枚举第一个数 a（下标 first）
    // a 最多枚举到倒数第三个位置（因为还要留b、c）
    for first := 0; first < n; first++ {
        // -------------------- 去重 --------------------
        // 如果当前数和上一个数相同，跳过，避免重复三元组
        if first > 0 && nums[first] == nums[first-1] {
            continue
        }

        // 目标：在 a 后面找 b + c = -a
        // 所以定义 target = -a
        target := -1 * nums[first]

        // 第三个数 c 的指针，初始指向数组最右端
        third := n - 1

        // 第二层循环：枚举第二个数 b（下标 second）
        // b 一定在 a 后面：second = first + 1
        for second := first + 1; second < n; second++ {
            // -------------------- 去重 --------------------
            // 如果当前 b 和上一个 b 相同，跳过，避免重复
            if second > first+1 && nums[second] == nums[second-1] {
                continue
            }

            // -------------------- 双指针核心 --------------------
            // 如果 b + c 太大，就把 c 往左移（变小）
            // 保证 second < third，指针不能相遇
            for second < third && nums[second]+nums[third] > target {
                third--
            }

            // -------------------- 终止条件 --------------------
            // 如果 b 和 c 指针重合了，说明后面不可能找到解
            if second == third {
                break
            }

            // -------------------- 找到答案 --------------------
            // 如果 b + c == -a  →  a + b + c = 0
            if nums[second]+nums[third] == target {
                // 把三元组加入结果
                ans = append(ans, []int{nums[first], nums[second], nums[third]})
            }
        }
    }

    // 返回所有不重复的三元组
    return ans
}
```
